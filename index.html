        function shuffleArray(array) {

            for (let i = array.length - 1; i > 0; i--) {

                const j = Math.floor(Math.random() * (i + 1));

                [array[i], array[j]] = [array[j], array[i]];

            }

        }

        function renderQuiz() {

            const wrapper = document.getElementById('questions-wrapper');

            wrapper.innerHTML = '';

           

            questions.forEach((item, index) => {

                const card = document.createElement('div');

                card.id = `q-card-${index}`;

               

                let imgHtml = item.img ? `<div class="my-4 flex justify-center bg-slate-50 p-4 rounded border border-dashed border-gray-300">${item.img}</div>` : '';

               

                let optionsHtml = '';

                item.a.forEach((opt, optIdx) => {

                    optionsHtml += `

                card.innerHTML = `

                    <div class="mb-2 flex items-center justify-between text-xs font-bold text-blue-900 tracking-wide uppercase">

        function toggleHint(index) {

            const box = document.getElementById(`hint-box-${index}`);

            box.classList.toggle('hidden');

        }

        function selectOption(qIdx, optIdx) {

            userAnswers[qIdx] = optIdx;

            document.getElementById('action-feedback').innerText = "Respuesta registrada temporalmente.";

            setTimeout(() => { document.getElementById('action-feedback').innerText = ""; }, 1200);

        }

        function move(step) {

            document.getElementById(`q-card-${currentIndex}`).classList.remove('active');

            currentIndex += step;

            if(currentIndex < 0) currentIndex = 0;

            if(currentIndex >= questions.length) currentIndex = questions.length - 1;

            document.getElementById(`q-card-${currentIndex}`).classList.add('active');

            updateNavigation();

        }

        function updateNavigation() {

            const total = questions.length;

            document.getElementById('progress-text').innerText = `Pregunta ${currentIndex + 1} de ${total}`;

            const pct = Math.round(((currentIndex + 1) / total) * 100);

            document.getElementById('progress-bar').style.width = `${pct}%`;

            document.getElementById('percentage-text').innerText = `${pct}%`;

            // Control de botones

            document.getElementById('btn-prev').classList.toggle('hidden', currentIndex === 0);

           

            if(currentIndex === total - 1) {

                document.getElementById('btn-next').classList.add('hidden');

                document.getElementById('btn-submit').classList.remove('hidden');

            } else {

                document.getElementById('btn-next').classList.remove('hidden');

                document.getElementById('btn-submit').classList.add('hidden');

            }

        }

        function finish() {

            let unanswered = 0;

            questions.forEach((item, index) => {

                if(userAnswers[index] === null) unanswered++;

                else if(userAnswers[index] === item.c) correctCount++;

            });

            if(unanswered > 0) {

                if(!confirm(`Tiene ${unanswered} preguntas sin responder en su hoja de respuestas. ¿Desea finalizar la evaluación técnica de todos modos?`)) {

                    return;

                }

            }

            document.getElementById('quiz-container').classList.add('hidden');

            const resDiv = document.getElementById('results');

            resDiv.classList.remove('hidden');

            const scoreVal = document.getElementById('score-val');

            scoreVal.innerText = `${correctCount} / ${questions.length}`;

            const pctScore = (correctCount / questions.length) * 100;

            const badge = document.getElementById('badge');

            const desc = document.getElementById('desc-val');

            if(pctScore >= 90) {
